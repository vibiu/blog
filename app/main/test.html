def bubble_sort(bubble_list):
    list_length = len(bubble_list)
    for i in range(list_length):
        for j in range(i, list_length):
            if bubble_list[i] > bubble_list[j]:
                bubble_list[i],bubble_list[j] = bubble_list[j],bubble_list[i]
    return bubble_list
